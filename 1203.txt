#include <iostream>
using namespace std;
int cnt=0;
void merge_sort(int *A,int x,int y, int *T)
{
	if(y-x>1)
	{
		int m=x+(y-x)/2;
		int p=x,q=m,i=x;
		merge_sort(A,x,m,T);
		merge_sort(A,m,y,T);
		while(p<m||q<y)
		{
			if(q>=y||(p<m && A[p]<A[q]))  T[i++]=A[p++];
			else  
			{
				T[i++]=A[q++];
				cnt+=m-p;
			}
		}
		for(i=x;i<y;i++)
			A[i]=T[i];
	}
}
int main()
{
	int n; 
	int A[50010];
	int T[50010];
	while(cin>>n)
	{
		int k;  
		for(k=0;k<n;k++)
		{
			cin>>A[k];
		}
		merge_sort(A,0,n,T);
		cout<<cnt<<endl;
		cnt=0;
	}
	return 0;
}
