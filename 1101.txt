#include <iostream> 
using namespace std; 
int main() 
{
	char grid[15][15];
	int n, kase=0;
	while (cin>>n) 
	{
		int i,j;
		for(i=0; i<n;i++) 
			cin>>grid[i];
		for(i=0; i<n;i++)
			for(j=0; j<n;j++) 
				if(grid[i][j]=='.')
				{
					for(char ch='A'; ch<='Z'; ch++)
					{ 		
						bool ok=true;
						if(i>0 && grid[i-1][j]==ch) 	
							ok=false; 
						if(i<n-1 && grid[i+1][j]==ch) 
							ok=false;
						if(j>0 && grid[i][j-1]==ch) 
							ok=false;
						if(j<n-1 && grid[i][j+1]==ch) 
							ok=false;
						if(ok) 
						{ grid[i][j]=ch; break; } 
					}
				}
			kase++;  
			if(kase>1)   cout<<endl;
			for(i=0; i<n; i++)
				cout<<grid[i]<<endl;
	}  
	return 0;
}
