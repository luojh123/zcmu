#include <string.h>
#include <stdio.h>
int main()
{
    int n,i;
    while(~scanf("%d",&n))
    {
        char s[5];
        for(i=0;i<n;i++)
        {
            scanf("%s",s);
            if(strlen(s)==5) printf("%d\n",3);
            else if( (s[0]=='o'&& s[1]=='n') ||( s[0]=='o'&& s[2]=='e')||(s[1]=='n'&& s[2]=='e') )
                printf("%d\n",1);
            else
                printf("%d\n",2);
        }
    }
    return 0;
}
