#include <iostream>
#include <iomanip>
using namespace std;
int main( ) 
{     
	int n,i,a,j,t;     
	float sum;      
	cin>>n;      
	for(i=1;i<=n;i++)  
	{         
		cin>>a;     
		sum=0;  t=1;    
		for(j=1;j<=a;j++)   
		{
			sum=sum+t*(1.0/j);   
			t=-t;
		}
		cout<<setiosflags(ios::fixed)<<setprecision(3)<<sum;
		cout<<endl;
	} 
}
