#include<cstdio>
int f(long long x)
{
    int sum=0;
    while(x > 0)
    {
        sum += x%10;
        x = x/10; 
    }  
    return sum;
}
 
int main()
{
    long long  a,b;
    while(scanf("%lld%lld",&a,&b)!=EOF)
    {
        int sum;
        sum=f(a)+f(b);
        printf("%d\n",sum);
    }
    return 0;
}
