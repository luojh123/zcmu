#include<iostream>
using namespace std;
int main()
{
    int T,n,ai,bi;
    cin>>T;
    while(T--)
    {
        int count=0;
        cin>>n;
        while(n--)
        {
            cin>>ai>>bi;
            if( (bi-ai)>=2 )
                count++;
        }
        cout<<count<<endl;
    }
    return 0;
}
