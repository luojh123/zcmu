#include <algorithm>
#include <iostream>
using namespace std;

const int maxn=1000000+10;
long long a[maxn],c[maxn],t,m;
int main()
{
	int n,i;
	while(cin>>n)
	{
		t=0;
		for(i=1;i<=n;i++)
		{
			cin>>a[i];
			t+=a[i];
		}
		m=t/n;
		c[0]=0;
		for(i=1;i<n;i++)
			c[i]=c[i-1]+a[i]-m;  //递推C数组
		sort(c,c+n);
		long long x1=c[n/2],ans=0;  //计算x1
		for(i=0;i<n;i++)
			ans+=abs(x1-c[i]);
		cout<<ans<<endl;
	}
	return 0;
}
