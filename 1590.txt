#include <set>
#include <iostream>
using namespace std;
int main()
{
	int n,k,tmp,ans;
	while(cin>>n)
	{
		set <int> s;
		for(int i=1;i<=n;i++)
		{
			cin>>tmp;
			s.insert(tmp);
		}
		cin>>k;
		if(n==s.size())
		{
			ans=n-k;   //刚好，就直接减 
		}
		else
		{
			tmp=n-s.size();//可以不减少的前提下，提供的最大数量 
			if(tmp>=k)//够减，，数量不变 
			ans=s.size();
			else      //不够减，减去多余的后，再减种类数 
			ans=s.size()-(k-tmp);
		}
		cout<<ans<<endl;
	}
	return 0;
}
