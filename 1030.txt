#include<iostream> 
#include<cstring>   //为了使用memset函数
using namespace std; 
			
int main() 
{
  int n, x, c[101];
  while(cin>>n && n) 
  {
	  int i,j;
    memset(c, 0, sizeof(c));
    for(i=0;i<n;i++) 
	{
      cin>>x;
      c[x]++;
    }
    int first=1; 	//标志first=1表示还没有输出过整数
    for(i=1;i<=100;i++)
      for(j=0;j<c[i];j++) 
	  {
        if(!first) cout<<" ";  //从第二个数开始，每输出一个数之前先输出一个空格
        first=0;
        cout<<i;
      }
    cout<<endl;
  }
  return 0;
}
