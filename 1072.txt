#include <iostream>
#include <iomanip>
using namespace std;
int main()
{
	int n,i,a;
	double price;
	cin>>n;
	for(i=1;i<=n;i++)
	{
		cin>>a;
		switch(a) 
		{
		    case 1:price=3;break;
		    case 2:price=2.5;break;
		    case 3:price=4.1;break;
		    case 4:price=10.2;break;
		    default:price=0;break;
		}
		if (price==0) cout<<"price=0.00\n";
		else 
		{	
			cout<<"price=";
			cout<<setiosflags(ios::fixed)<<setprecision(2)<<price;
			cout<<endl;
		}
	}
	return 0;
}
