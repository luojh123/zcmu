#include <algorithm>
#include<vector>
#include<iostream>
using namespace std;
int main()
{
    int T,n,i,x;
    vector<int> v; 
    cin>>T;
    while(T--)
    {
        v.clear();
        cin>>n;
        for(i=0;i<n;i++)
        {
            cin>>x;
            v.push_back(x);
        }
        int maxi=*max_element(v.begin(),v.end());
        int mini=*min_element(v.begin(),v.end());
        cout<<maxi-mini<<endl;
    }
    return 0;
}
