#include <stdio.h>
int main()
{
    int n,m,i;
    while(~scanf("%d%d",&n,&m))
    {
        int L,R,x,avg;
        int *sum=new int[n+5];
        sum[0]=0;
        for(i=1;i<=n;i++)
        {
            scanf("%d",&x);
            sum[i]=sum[i-1]+x;
        }
        for(i=0;i<m;i++)
        {
            scanf("%d%d",&L,&R);
            avg=( sum[R]-sum[L-1] )/(R-L+1);
            printf("%d\n",avg);
        }
     
    }
 
    return 0;
}
