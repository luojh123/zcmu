#include <iostream>
#include <functional> 
#include <algorithm>
#include<stdio.h>
#include<string>
using namespace std;
int main()
{	
	int n,i;
	while(cin>>n)
	{
		int a[100];
		char s[100];
		string s1,s2;
		for(i=0;i<n;i++)
			cin>>a[i];
		sort(a,a+n,greater<int>());
		for (i=0;i<n;i++)
		{
			sprintf(s,"%d",a[i]);
			s1=s1+s+" ";
		}
		int len=s1.length();
		s2=s1.substr(0,len-1);
		cout<<s2<<endl;
	}
	return 0;
}
