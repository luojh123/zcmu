#include<vector>
#include<iomanip>
#include<iostream>
using namespace std;
 
int main()
{
    int n,i;  
    while(cin>>n)
    {
        vector<double> si(n+1),ci(n+1);
        vector<char> vc(n+1);  
        double a=0,b=0,c;
        for(i=0;i<n;i++)
        {
            cin>>si[i]>>vc[i];
            switch (vc[i])
            {
                case 'A': ci[i]=4.0;break;
                case 'B':  ci[i]=3.0;break;
                case 'C':  ci[i]=2.0;break;
                case 'D':  ci[i]=1.3; break;
                case 'F':   ci[i]=0; break;
                default : si[i]=ci[i]=0 ;break;
            }
            a=a+si[i]*ci[i];  b=b+si[i];
        }
        if (a==0 || b==0)  c=0;
        else    c=a/b;
     cout<<setiosflags(ios::fixed)<<setprecision(2)<<c<<"\n";
    }
    return 0;
}
