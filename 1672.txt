#include <stdio.h>
int main()
{
    int n,m,i,j; 
    while (~scanf("%d%d", &n,&m))
    {
        int a[4]={0,0,0,0};
        int count=0;
        for(i=0;i<n;i++)
        {
            for(j=0;j<m;j++)
            {   
                scanf("%d%d%d", &a[0],&a[1],&a[2]);
                if( (a[0]+a[1]+a[2])>1 )
                    count++;
            }
        }
        printf("%d\n", count);
    }
    return 0;
}
