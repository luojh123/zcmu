#include<algorithm>
#include<iostream>
using namespace std;
int main()
{
    int n,k,i;  
    while(cin>>n>>k)
    {
        int x;
        int *a=new int[n+1];
        for(i=0;i<n;i++)
        {
            cin>>x;
            a[i]=x;
        }
        sort(a,a+n);
        cout<<a[k-1]<<endl;
    }
    return 0;
}
