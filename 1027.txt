#include<iostream>     
using namespace std; 
int gcd(int n, int m)//求2个数的最大公约数
{
	int t;
    if (n<m)
    {
        t=n; n=m; m=t;
    }
    if (m==0) return n;
    return gcd(m,n%m);
}
int n_gcd(int a[], int n)//求n个数的最大公约数
{
    int c=gcd(a[0],a[1]);
    int i;
    for (i=2; i<n; i++)
    {
        c=gcd(c,a[i]);
    }
    return c;
}
int main()
{
    int a[100];
    int i,n,r;
	cin>>r;
	while(r--)
	{
		cin>>n;
		for(i=0;i<n;i++)
			cin>>a[i];
		cout<<n_gcd(a,n)<<endl;
	}
    return 0;
}
