#include <iostream>
using namespace std;
int main()
{
    int n, m;
    while (cin>>n>>m)
    { 
        int num0, num1,i;
        if (m>n+1)
        {
            cout<<"NO\n";
            continue;
        }
        num0=n-m+1;
        if (m==0)
            num0--;
        num1=m;
        if (num0>0)
        {
            cout<<"0";
            num0--;
        }
        for (i=0;i<num1;i++)
        {
            if (i==0)
                cout<<"1";
            else
                cout<<"01";
        }
        for (i=0;i<num0;i++)
            cout<<"0";
        cout<<endl;
    }
    return 0;
}
