#include <algorithm>
#include <string>
#include <iostream>
using namespace std;
 
struct Stu {
  string name;
  int grade;
  bool operator < (const Stu& x) const { 
      if(grade==x.grade)
          return name<x.name;
      else return grade>x.grade;
  }
};
 
 
int main()
{
    int n,i;
    while(cin>>n)
    {
        Stu *a=new Stu[n+1];
        for(i=0;i<n;i++)
        {
            cin>>a[i].name>>a[i].grade;
        }
        stable_sort(a,a+n);
        for(i=0;i<n;i++)
        {
            cout<<a[i].name<<" "<<a[i].grade;
            cout<<endl;
        }
    }
 
    return 0;
}
