#include<iostream>
#include<cstdio>
#include<cmath>
using namespace std;
 
int main()
{
    int t,i,n;
    cin>>t;
    while(t--)
    {
       cin>>n;
       for(i=sqrt(n);i>0;i--)
       {
          if(n%i==0&&(n/i-i)%2==0&&n/i!=i)
          {
              cout<<(n/i-i)/2<<endl;
              break;
          }
       }
       if(i==0) cout<<-1<<endl;
    }
    return 0;
}
