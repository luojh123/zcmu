#include <cstring>
#include <iostream> 
using namespace std;  
int main () 
{
	char word[100];
	while(cin>>word)
	{
		int len=strlen(word);
		for (int i=1;i<=len;i++)
			if(len%i==0)
			{
				int ok=1;
				for(int j=1;j<len;j++)
				{
					if(word[j]!=word[j%i])
					{
						ok=0;break;
					}
				}
				if(ok)
				{
					cout<<i<<endl; break;
				}
			}
	}
	return 0;   
}
