#include <iostream> 
#include <cstring> 
using namespace std; 
int num[8]={128, 64, 32, 16, 8, 4, 2, 1}; 
int main() 
{ 
    int N;  
    cin>>N;
    while(N--) 
    { 
        char s[40]; 
        cin>>s;
        int length=strlen(s); 
        int k=0; 
        for(int i=0; i<length;i++) 
        { 
            if(i%8==0 && i) 
            { 
                cout<<k<<'.'; 
                k=0; 
            } 
            k=k+(s[i]-'0')*num[i%8]; 
        } 
        cout<<k<<endl; 
    } 
    return 0; 
}
