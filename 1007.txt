#include <iostream>  
#include<iomanip>
using namespace std;   
int main()  
{  
    int n,a;  
    while(cin>>n&&n)  
    {  
		double sum=0; 
		a=n; 
        while(n--)  
        {  
            int i;  
            cin>>i;  
            sum+=i;  
        }  
		cout<<setiosflags(ios::fixed)<<setprecision(2);
        cout<<sum/a<<endl;  
    }   
    return 0;  
}
