#include<vector>
#include<cmath>
#include<algorithm>
#include <iostream> 
using namespace std; 
int cmp(int a,int b)
{
    return abs(a)>abs(b);
}
int main() 
{
    int n,i;
    while (cin>>n && n) 
    {
        vector<int> v;
        int x;
        for(i=0;i<n;i++)
        {
            cin>>x;
            v.push_back(x);
        }
        sort(v.begin(),v.end(),cmp);
        cout<<v[0];
        for(i=1;i<n;i++)
        {
            cout<<" "<<v[i];
        }
        cout<<endl;
    }  
    return 0;
}
