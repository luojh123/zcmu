#include <iostream> 
#include <string.h>
using namespace std;  
int main () 
{
	char a[100];
	int i,j,k,count,b[100];
	 while (cin>>a)
	 {
		 k=0;
		for(i=0;i<strlen(a);i++)
		{
			b[i]=a[i];
		}
		for(i=0;i<strlen(a);i++)
		{
			if (b[i]==0)  continue;
			for(j=i+1;j<strlen(a);j++)
			{				
				if (b[i]==b[j]) b[j]=0;
			}
		}
		for(i=0;i<strlen(a);i++)
		{
			if (b[i]==0) k++;
		}
		count=strlen(a)-k;
		if (count%2==0)  cout<<"CHAT WITH HER!"<<endl;
		else cout<<"IGNORE HIM!"<<endl;
	 }
	return 0;   
}
