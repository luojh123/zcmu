#include <iostream>
using namespace std;
int prime(int n)
{
    int i;
    for(i=2;i*i<=n;i++)
    {
        if(n%i==0)   
            return 0;
    }
    return 1;
}
 
int main()
{
    int n,i,j,f=0;
    while(cin>>n && n)
    {
        for(i=3;i<=n/2;i+=2)
        {
            j=n-i;
            if(prime(j) && prime(i))
            {
                cout <<n<<" = "<<i<<" + "<<j<<endl;
                f=1;break;
            }
        }
	if(!f)  cout<<"Goldbach's conjecture is wrong."<<endl;
    }
    return 0;
}
