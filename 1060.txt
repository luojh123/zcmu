#include<cmath>
#include<algorithm>
#include<iostream>
using namespace std;
 
int main()
{
	int n,c,x[210];
	while(cin>>n)
	{
		int i,sum=0;
		x[0]=0;
		for(i=1;i<=n;i++)
		{
			cin>>x[i];
		}
		sort(x,x+(n+1));
		for(i=0;i<=n;i++)
		{
			sum+=abs(x[i]-x[(n+1)/2]);
		}
		cin>>c;
		sum=sum*c;
		cout<<sum<<endl;
	}
    return 0;
}
