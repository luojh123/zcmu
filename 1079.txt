#include <iostream>
#include <iomanip>
#include <cmath>
using namespace std;
int main( ) 
{     
	int n,i,t;     
	float sum,eps,j,a;      
	cin>>n;      
	for(i=1;i<=n;i++)  
	{         
		cin>>eps;     
		sum=0; t=1;j=1;a=1;   
        while (fabs(a)>=eps)
			{ 
				sum=sum+a;
				j=j+2;
				t=-t;
				a=t/j;
			}
		cout<<setiosflags(ios::fixed)<<setprecision(6)<<sum;
		cout<<endl;
	} 
	return 0;
}
