#include <cstring>
#include <iostream>
using namespace std;
#define N 1000
int a[N];
void fun(int n) 
{      
	int i,j,k,count,temp;
	memset(a,0,sizeof(a));
    a[0]=1;
    count=1;
	for(i=1;i<=n;i++)
	{
		k=0;
		for(j=0;j<count;j++)
		{
			temp=a[j]*i+k;
			a[j]=temp%10;
			k=temp/10;    
		}
		while(k)//记录进位    
		{
			a[count++]=k%10;
			k/=10;
		}
	}
	for(j=N-1;j>=0;j--)      
		if(a[j])
			break;//忽略前导0
		for(i=count-1;i>=0;i--)
			cout<<a[i];
		cout<<endl;
}
int main()
{
    int n,i,a;
    cin>>n;
    for (i=1;i<=n;i++)
    {
        cin>>a;
        fun(a);
    }
    return 0;
}
