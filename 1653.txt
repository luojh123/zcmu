#include<cstring>
#include <iostream>
using namespace std;
 
int main()
{
    int a[105];
    int n,k,i,x;
    while(cin>>n>>k)
    {
        int t=n/k;
        int sum=0;
        memset(a,0,sizeof(a)); 
        for(i=0;i<n;i++)
        {
            cin>>x;
            a[x]++;
        }
        for(i=0;i<105;i++)
        {
            if(a[i]>t)
                sum=sum+(a[i]-t);
        }
        cout<<sum<<endl;
    }
    return 0;
}
