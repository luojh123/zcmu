#include <iostream>
using namespace std;
 
int main()
{
    int n,k=1; 
    while (cin>>n && n)
    {
        int *a=new int[n];
        int i,sum=0,count=0; 
        for (i=0;i<n;i++)
        {
            cin>>a[i];
            sum+=a[i];
        }
        int avg=sum/n;
        for (i=0;i<n;i++)
        {
            if(a[i]<avg)
                count=count+(avg-a[i]);
        }
        cout<<"Set #"<<k++<<endl;
        cout<<"The minimum number of moves is "<<count<<"."<<endl;
        cout<<endl;
    }
 
    return 0;
}
