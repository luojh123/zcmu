#include <iostream>
#include <algorithm>
using namespace std; 
 
int main()
{    
    int p[8],n;
    while(cin>>n)
    {
        int i;
        for(i=0; i<n; i++)
        {  
            cin>>p[i];
        }   
        sort(p,p+n);
        int f=1;
        for(i=0; i<n; ++i)
        {       
            if(f)
            {
                cout<<p[i]; f=0;
            }
            else cout<<" "<<p[i];
        }
        cout<<endl;
        while( next_permutation(p,p+n) )    
        {
             f=1;
            for(i=0; i<n; ++i)
            {
                if(f)
                {
                    cout<<p[i]; f=0;
                }
                else cout<<" "<<p[i];
            }
                cout<<endl;    
        }
    }
    return 0;
}
