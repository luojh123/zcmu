#include<string.h>
#include<iostream>  
using namespace std;  
int a[60],flag[1010]; 
int main()  
{  
	
    int i,m,n,now,max;  
    int cases=0;  
    while(cin>>m>>n && (m||n))  
    {  
        cases++;  
        now=max=0; 
        memset(flag,0,sizeof(flag)); 
        for(i=0;i<n;i++)  
        {  
            cin>>a[i];  
            flag[a[i]]=1;  
        }  
        for(i=m*n;i>0;i--)  
        {  
            if(flag[i])  
            {  
                now++;  
                if(now>max)  
                    max=now;  
            }  
            else  
                now--;  
        }  
		cout<<"Case "<<cases<<": "<<max<<endl;  
    }  
	return 0;
}
