#include <iostream>
using namespace std;
int weight[105];
int value[105];
int dp[1000];
void Solve( int n, int w )
{   
    for( int i=1; i<=n; i++ )
    {
        for( int j=w; j>=weight[i]; j-- )
        {           
            if( dp[j-weight[i]] + value[i] > dp[j] )
                dp[j] = dp[j-weight[i]] + value[i];         
        }
    }
    cout << dp[w] << endl;
}
 
int main()
{
    int n, w;
    while(  cin >> w>>n)
    {       
        for( int i=1; i<=n; i++ )
        {
            cin >> weight[i] >> value[i];
        }
        Solve( n, w );  
    }
    return 0;
}
