#include<cmath>
#include<iostream>
using namespace std;
int isprime(int n)
{
      if (n == 1)  return 0;
    for(int i=2;i<= (int) sqrt(n);i++)
    {
        if (n%i==0) return 0;
    }
    return 1;
}
int main()
{
    int n,i,a[15]; 
    while(cin>>n)
    {
        for(i=0;i<n;i++)
        {
            cin>>a[i];
            if(isprime(a[i])) cout<<"YES"<<endl;
            else cout<<"NO"<<endl;
        }
    }
    return 0;
}
