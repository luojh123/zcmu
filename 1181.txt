#include<iostream>
#include<string>
#include<cstring>
using namespace std;
#define MAX 10010
int a[MAX], b[MAX];
int main()
{
    int N, len1, len2, i, k, up;
    string str1, str2;
    cin>>N; 
    while(N--)
    {
        memset(a,0, sizeof(a));
        memset(b,0, sizeof(b));
        cin>>str1>>str2;
        len1 = str1.length(); len2 = str2.length();
        for(i = len1 - 1, k = 0; i >= 0; --i)
            a[k++] = str1[i] - '0';
        for(i = len2 - 1, k = 0; i >= 0; --i)
            b[k++] = str2[i] - '0';
        for(i = 0, up = 0; i < MAX; ++i)
        {
            a[i] = a[i] + b[i] + up;
            up = a[i] / 10;
            a[i] %= 10;     
        }
        for(i = MAX -1; i >= 0; --i)
            if(a[i]) break;
        if(i==-1) cout<<0;
        else
        {
            for(k = i; k >= 0; --k)
                cout<<a[k];
        }
        cout<<endl;
    }
    return 0;
}
