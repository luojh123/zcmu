#include<string>
#include<stack>
#include<iostream>
using namespace std;
int main()
{	
	int r;
	cin>>r;
	while(r--)
	{
		string c;
		cin>>c;
		int a,i,sum=0;
		for(i=0;i<c.length();i++)
		{
			a=c[i];
			sum+=a;
		}
		stack<int> s;
		while(sum)
		{
			s.push(sum%2);
			sum/=2;
		}
		while(!s.empty())
		{
			cout<<s.top();
			s.pop();
		}
		cout<<endl;
	}
	return 0;
}
