#include<stdio.h> 
#include<stdlib.h> 
int main() 
{ 
    int t,i; 
    long long a[51]; 
    a[1]=1;a[2]=2;a[3]=3; 
    for(i=3;i<51;i++) 
        a[i]=a[i-1]+a[i-2]; 
    scanf("%d",&t); 
    while(t--) 
    {
        int m,n,i,j; 
        scanf("%d%d",&m,&n); 
        j=n-m; 
        printf("%lld\n",a[j]); 
    } 
    return 0; 
}
