#include<cstring>
#include <iostream>
using namespace std;
 
int main()
{
    int a[1005];
    int n,i,x;
    while(cin>>n)
    {
        memset(a,0,sizeof(a)); 
        for(i=0;i<n;i++)
        {
            cin>>x;
            a[x]++;
        }
        for(i=0;i<1005;i++)
        {
            if(a[i]==2)
            {
                cout<<i<<endl;
                break;
            }
        }
    }
    return 0;
}
