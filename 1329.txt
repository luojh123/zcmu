int main()
{
    char t[128],max;
    int i;

        for (max = i = 0 ; t[i] ; i++)
        {
            if (t[i] > max)
                max = t[i];
        }
        for (i = 0 ; t[i] ; i++)
        {
			printf("%c", t[i]);
            if (t[i] == max)
                printf("%s", "(max)");

    }

}
