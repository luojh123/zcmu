#include <iostream>  
using namespace std;  
int main()  
{  
    int a,m,n,mm,nn;  
    while(cin>>m)  
    {  
        cin>>n>>a;  
        mm=m/a;  
        if(m%a!=0)  
        mm++;  
       nn=n/a;  
       if(n%a!=0)  
       nn++;  
       cout<<mm*nn<<endl;  
    }        
    return 0;  
}
