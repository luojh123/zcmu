#include<algorithm>
#include<iostream>
using namespace std;
#define N 20010
int main()
{
	int a[N],b[N];
	int n,m,i;
	while(cin>>n>>m && n && m)
	{
		for(i=0;i<n;i++)  cin>>a[i];
		for(i=0;i<m;i++)  cin>>b[i];
		sort(a,a+n);
		sort(b,b+m);
		int cur=0,cost=0;
		for(i=0;i<m;i++)
			if(a[cur]<=b[i])
			{
				cost+=b[i];
				cur++;
				if(cur==n) break;
			}
		if (cur<n)  cout<<"Loowater is doomed!"<<endl;
		else cout<<cost<<endl;
	}
	return 0;
}
