int mod_exp(int a, int b, int c)  //¿ìËÙÃÝÈ¡Óàa^b%c
{
    int res, t;
    res = 1 % c; 
    t = a % c;
    while (b)
    {
        if (b % 2 == 1)
        {
            res = res * t % c;
        }
        t = t * t % c;
        b = b/2;
    }
    return res;
}
