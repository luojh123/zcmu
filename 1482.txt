#include <vector>
#include <algorithm>
#include <iostream>
using namespace std;
int main()
{
	int n;
	while(cin>>n)
	{
		vector <int> v;
		int max,min;
		int s;
		while(n--)
		{
			cin>>s;
			v.push_back(s);
		}
		max=*max_element(v.begin(),v.end());
		min=*min_element(v.begin(),v.end());
		cout<<max<<" "<<min<<endl;
	}
	return 0;
}
