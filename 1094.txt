#include <string.h>
#include <iostream>
using namespace std;
int main() 
{
    char string[100];
    int i,L,count;
	int r;
	cin>>r;
	while(r--)
	{
		count=0;
		cin>>string;
		L=strlen(string);
		if (string[0]=='-') 
		{
			for (i=1;i<=L-1;i++) 
				if(string[i]=='2')
					count++;
		}
		else 
		{
			for (i=0;i<=L-1;i++) 
				if(string[i]=='2')
					count++;
		}
		cout<<"count="<<count<<endl;

	}
	  return 0;
}
