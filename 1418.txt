#include <stdio.h>
int main()
{
    int n,i,a[3];
    while(~scanf("%d",&n))
    {
        int c=0;
        for(i=0;i<n;i++)
        {
            scanf("%d%d%d",&a[0],&a[1],&a[2]);
            if( (a[0]+a[1]+a[2])>=2 )
                c++;
        }
        printf("%d\n",c);
    }
 
    return 0;
}
