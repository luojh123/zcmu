#include<cstring>
#include<iostream>
using namespace std;

const int N=1010;
const int mod=100000007;

int dp[N][N+N][2];  //dp[i][j][z],i表示第i列，j表示i列内分成j部分，
                    //z表示第i列的两块巧克力处于分开的状态还合并的(0:分开,1:合并）
int main()
{

    int t,n,k;
    memset(dp,0,sizeof(dp));
    dp[1][1][1]=1;dp[1][2][0]=1;
    for(int i=2;i<N;i++)
	{
        for(int j=1;j<=i*2;j++)
		{
            dp[i][j][1]=(2*dp[i-1][j][0]+dp[i-1][j][1]+dp[i-1][j-1][0]+dp[i-1][j-1][1])%mod;
            dp[i][j][0]=(dp[i-1][j][0]+2*dp[i-1][j-1][1]+2*dp[i-1][j-1][0]+dp[i-1][j-2][0]+dp[i-1][j-2][1])%mod;
        }
    }
    cin>>t;
    while(t--){
        cin>>n>>k;
        cout<<(dp[n][k][1]+dp[n][k][0])%mod;
		cout<<endl;
    }
    return 0;
}
