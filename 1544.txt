#include<stdio.h>
#include <iostream>
using namespace std;
int main()
{
     char str[100];
     int n,i,j,num,word;
     cin>>n;
     for(j=0;j<n+1;j++)
     { 
         gets(str);
         char c;
         num=0;word=0;
         for(i=0;(c=str[i])!='\0';i++)
         {
             if(c==' ')  word=0;
             else if(!word)
             {
                 word=1;
                 num++;
             }
         }
         if(num)  cout<<num<<endl;
     }
     return 0;
}
