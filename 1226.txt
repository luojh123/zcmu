#include <iostream>
using namespace std;
int num[1000010];
int main()
{
    int T;
    cin>>T;
    while(T--)
    {   
        int i,n,max,temp;
        cin>>n;
        for (i=0;i<n;i++) 
            cin>>num[i];
        max=temp=num[0]+num[1];     
        for (i=2;i<n;i++) 
        { 
            if (temp > num[i-1])
                temp=temp+num[i];
            else temp=num[i-1] + num[i];
             
            if (max<temp) 
                max=temp;
        }
         
        if (max>0) cout<<max;
        else cout<<"Game Over";
        cout<<endl;
    }
    return 0;
}
