#include <iostream>
using namespace std;
int main( ) 
{     
    int n,i,a,sum;     
    cin>>n;      
    for(i=1;i<=n;i++)  
    {     
        sum=0;
        while(cin>>a)
        {
            if (a<=0) break;
            if (a%2!=0)
            sum=sum+a;
        }
        cout<<sum<<endl;
    } 
    return 0;
}
