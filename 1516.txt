#include <iostream>
#include <cctype>
#include <vector>
#include <string>
using namespace std;
int main()
{
    int n;
    cin>>n;
    while(n--)
    {
        string s;
        cin>>s;
        vector<char> v,u,w;
        int i,k,l;
        v.push_back(s[0]);
        for(i=1;i<s.size();i++)
        {
            if(!isupper(s[i]))
                v.push_back(s[i]);
            else
            {
                k=i;
                break;
            }
        }
        u.push_back(s[k]);
        for(i=k+1;i<s.size();i++)
        {
            if(!isupper(s[i]))
                u.push_back(s[i]);
            else
            {
                l=i;
                break;
            }
  
        }
        for(i=l;i<s.size();i++)
            w.push_back(s[i]);
        if(v==u&&u==w)
            cout<<"No"<<endl;
        else if(u==w)
            cout<<"Yes"<<endl;
        else
            cout<<"No"<<endl;
    }
    return 0;
}
