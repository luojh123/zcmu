#include <string.h>
#include <iostream>
using namespace std;
int main() 
{
    char string[100];
    int i,number,sum;
	int r;
	cin>>r;
	while(r--)
	{
		number=0;sum=0;
		cin>>string;
		number=strlen(string);
		if (string[0]=='-') 
		{
			number--;
			for (i=number;i>=1;i-- ) 
			{
				sum=sum+(string[i]-48);
			}
		}
		else 
		{
			for (i=number-1;i>=0; i-- ) 
			{
				sum=sum+(string[i]-48);
			}
		}
		cout<<"number="<<number<<",sum="<<sum<<endl;
	}
	  return 0;
}
