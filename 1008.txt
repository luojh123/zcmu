#include <iostream>  
using namespace std;   
int main()  
{  
    int i,j,n,m,a;  
    cin>>n;
    while(n--)
    {
        j=0;
        cin>>m>>a;
        for(i=1;i<=m*m;i++)
        {
                         
            cout<<a;  
            j++;
            if(j%m==0)  
                cout<<endl;
        }
		if(n!=0) cout<<endl;
    }
    return 0;  
}
