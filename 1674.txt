#include <iostream>
using namespace std;
 
int main()
{
    long long x,y,z;
    long long a,b,c;
      while(cin>>x>>y>>z)
      {
          a=(x+y+2*z);
          b=(x+y-2*z);
          c=(x-y);
          if(a<0 || b<0 || c<0)
              cout<<"wangwangwang\n";
          else if ( a%4 || b%4 || c%2 )
               cout<<"wangwangwang\n";
          else cout<<a/4<<" "<<b/4<<" "<<c/2<<endl;
      }
    return 0;
}
